//Set the icon item
setIcon(<woot:anvil>)

//Set the title String
setTitle("Blacksmithing Time")

//Set the description String
setDescription("Craft a Stygian Anvil. Once made, you must set it on a magma block to use the anvil.")

//Completed advancement color change
//setCompletedIconColor(0, 255, 0)
//setCompletedTitleColor(0, 255, 0)
//setUncompletedTitleColor(185, 143, 44)

//Add a parent advancement. The syntax is "modid:folder/path/to/the/advancement/then/filename"
addParent("mfb:bookcrafting/wo_02_stygianore")

setRequiresParents()

//Set the position
setPos(-173,-46)

//Hide the connection lines, so I can have a pretty shape
hideLines()

//Adds criteria with the trigger type. This function returns the criteria as an object
criteria = addCriteria("wootAnvil", "minecraft:inventory_changed")

//Sets the item for the criteria. This is the item that needs to be crafted.
criteria.addItem(<woot:anvil>)

//By not setting anything about requirements here, all criteria are automatically required
